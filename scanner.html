This scanner follows these rules

program ::= statement | program statement
statement ::= assignStmt | ifStmt
assignStmt ::= id = expr ;
ifStmt ::= if ( expr ) stmt
expr ::= id | int
id ::= (any number of lowercase and uppercase english alpahbets)
int ::= (any number of digits)
